Python еҶ…ж ёжәҗз Ғи§ЈжһҗпјҡйқһзңҹйқһеҒҮпјҢйӮЈд№ҲжҳҜд»Җд№Ҳпјҹ вҖӢ

иғҢжҷҜ вҖӢ

дёҠдёҖзҜҮжҸҗеҲ° Python дёӯ if жқЎд»¶з»“жһңеҖјеҲӨж–ӯеҮәзҺ°дәҶйқһзңҹ (True) йқһеҒҮ (False) зҡ„еҲӨж–ӯпјҢе…·дҪ“д»Јз Ғи§ҒжҢҮд»ӨPOP_JUMP_IF_FALSEе’ҢPOP_JUMP_IF_TRUEгҖӮзңӢеҲ°иҝҷж ·зҡ„и®ҫи®ЎпјҢжҳҜдёҚжҳҜжңүзӮ№еӣ°жғ‘пјҢдёӢйқўе°ұи®©жҲ‘жқҘеҲҶжһҗдёҖдёӢгҖӮ

жҠҪиұЎж„Ҹд№үдёҠзЎ®е®һеҸӘжңүзңҹжҲ–еҒҮпјҢдҪҶжҳҜеңЁ Python дёӯпјҲе…¶е®ғејұзұ»еһӢиҜӯиЁҖд№ҹжҳҜеҗҢж ·зҡ„дёҖдёӘеҺҹзҗҶпјүпјҢTrue жҲ–иҖ… False еңЁеҲӨж–ӯеҖјжҳҜеҗҰзӣёзӯүж—¶пјҢ

Python дёӯеҜ№дәҺеҖјзҡ„еҲӨж–ӯпјҢзңҹжҲ–еҒҮжҳҜдёӨдёӘеҲҶзұ»пјҢеҲҶеҲ«з®ЎзҗҶдәҶдёҖзұ»еһӢзҡ„еҖјгҖӮжҜ”еҰӮ 0пјҢз©әеӯ—з¬ҰдёІпјҢз©әеҲ—иЎЁпјҢз©әеӯ—е…ёзӯүзӯүпјҢиҝҷдәӣеҖјйғҪеҒҮгҖӮеғҸе…¶е®ғй•ҝеәҰеӨ§дәҺ 0 зҡ„еӯ—з¬ҰдёІпјҢж•°з»„гҖҒеӯ—е…ёпјҢе…¶еҖјжҳҜзңҹпјҲйҷҗ if еҲӨж–ӯпјүгҖӮ

POP_JUMP_IF_FALSE е’Ң POP_JUMP_IF_TRUE жҢҮд»Өи§Јжһҗ вҖӢ

POP_JUMP_IF_FALSE вҖӢ

еӣһйЎҫдёҖдёӢжҢҮд»ӨPOP_JUMP_IF_FALSEпјҢеҰӮжһңж ҲйЎ¶е…ғзҙ дёәеҒҮпјҢеҲҷи·іиҪ¬еҲ°жҢҮе®ҡзҡ„дҪҚзҪ®пјҢеҗҰеҲҷз»§з»ӯжү§иЎҢдёӢдёҖжқЎжҢҮд»ӨгҖӮ

c
    case TARGET(POP_JUMP_IF_FALSE): {       // pop TOS and jump if false
        PREDICTED(POP_JUMP_IF_FALSE);
        PyObject *cond = POP();
        int err;
        if (Py_IsTrue(cond)) {      // еҰӮжһңжқЎд»¶дёәзңҹпјҢеҲҷдёҚи·іиҪ¬
            Py_DECREF(cond);
            DISPATCH();
        }
        if (Py_IsFalse(cond)) {     // еҰӮжһңжқЎд»¶дёәеҒҮпјҢеҲҷи·іиҪ¬еҲ°жҢҮе®ҡдҪҚзҪ®
            Py_DECREF(cond);
            JUMPTO(oparg);
            CHECK_EVAL_BREAKER();
            DISPATCH();
        }
        err = PyObject_IsTrue(cond);
        Py_DECREF(cond);
        if (err > 0)
            ;
        else if (err == 0) {
            JUMPTO(oparg);              // и·іиҪ¬еҲ°жҢҮе®ҡдҪҚзҪ®пјҢoparg жҳҜжҢҮе®ҡдҪҚзҪ®зҡ„еҒҸз§»йҮҸ
            CHECK_EVAL_BREAKER();
        }
        else
            goto error;
        DISPATCH();
    }

POP_JUMP_IF_TRUE вҖӢ

POP_JUMP_IF_TRUEжҢҮд»Өд»Јз ҒеҰӮдёӢпјҢеҰӮжһңж ҲйЎ¶е…ғзҙ дёәзңҹпјҢеҲҷи·іиҪ¬еҲ°жҢҮе®ҡзҡ„дҪҚзҪ®пјҢеҗҰеҲҷз»§з»ӯжү§иЎҢдёӢдёҖжқЎжҢҮд»ӨгҖӮ

    case TARGET(POP_JUMP_IF_TRUE): {        // pop TOS and jump if true
        PREDICTED(POP_JUMP_IF_TRUE);

        if (Py_IsFalse(cond)) {             // еҰӮжһңжқЎд»¶дёәеҒҮпјҢеҲҷдёҚи·іиҪ¬
            Py_DECREF(cond);
            DISPATCH();
        }
        if (Py_IsTrue(cond)) {              // еҰӮжһңжқЎд»¶дёәзңҹпјҢеҲҷи·іиҪ¬еҲ°жҢҮе®ҡдҪҚзҪ®
            Py_DECREF(cond);
            JUMPTO(oparg);
            CHECK_EVAL_BREAKER();
            DISPATCH();
        }
        err = PyObject_IsTrue(cond);
        Py_DECREF(cond);
        if (err > 0) {
            JUMPTO(oparg);                  // еҰӮжһңжқЎд»¶дёәзңҹпјҢеҲҷи·іиҪ¬еҲ°жҢҮе®ҡдҪҚзҪ®
            CHECK_EVAL_BREAKER();
        }
        else if (err == 0)
            ;
        else
            goto error;
        DISPATCH();
    }

иө·еҲ°е…ій”®дҪңз”Ёзҡ„жҳҜPy_IsTrueе’ҢPyObject_IsTrueгҖӮжҺҘдёӢжқҘе°ұеҲҶжһҗдёҖдёӢиҝҷдёӨеқ—еёҰд»Јз Ғзҡ„е®һзҺ°

Py_IsTrue жҳҜе®Ҹе®ҡд№ү вҖӢ

Py_Falseе’ҢPy_TrueжҳҜе…ЁеұҖе®ҡд№үзҡ„еҖјпјҢе®ғд»¬еҲҶеҲ«жҳҜ_Py_FalseStructе’Ң_Py_TrueStructз»“жһ„зҡ„еҶ…еӯҳең°еқҖпјҢзЁӢеәҸеҗҜеҠЁд№ӢеҗҺе®ғд»¬зҡ„еҶ…еӯҳең°еқҖжҳҜдёҚеҸҳзҡ„гҖӮPy_Is(x, y) е…¶е®һе°ұжҳҜжҜ”иҫғдёӨдёӘеҜ№иұЎзҡ„еҶ…еӯҳең°еқҖжҳҜеҗҰзӣёзӯүпјҢеҰӮжһңзӣёзӯүеҲҷиҝ”еӣһ 1пјҢеҗҰеҲҷиҝ”еӣһ 0гҖӮ

c
/* Use these macros */
#define Py_False ((PyObject *) &_Py_FalseStruct)
#define Py_True ((PyObject *) &_Py_TrueStruct)

// Test if an object is the True singleton, the same as "x is True" in Python.
PyAPI_FUNC(int) Py_IsTrue(PyObject *x);
#define Py_IsTrue(x) Py_Is((x), Py_True)

// Test if an object is the False singleton, the same as "x is False" in Python.
PyAPI_FUNC(int) Py_IsFalse(PyObject *x);
#define Py_IsFalse(x) Py_Is((x), Py_False)

// Test if the 'x' object is the 'y' object, the same as "x is y" in Python.
PyAPI_FUNC(int) Py_Is(PyObject *x, PyObject *y);
#define Py_Is(x, y) ((x) == (y))

/* The objects representing bool values False and True */

struct _longobject _Py_FalseStruct = {
    PyVarObject_HEAD_INIT(&PyBool_Type, 0)
    { 0 }
};

struct _longobject _Py_TrueStruct = {
    PyVarObject_HEAD_INIT(&PyBool_Type, 1)
    { 1 }
};

int
PyObject_IsTrue(PyObject *v)
{
    Py_ssize_t res;
    if (v == Py_True)           // еҰӮжһң v жҳҜ Py_TrueпјҢеҲҷиҝ”еӣһ 1
        return 1;
    if (v == Py_False)          // еҰӮжһң v жҳҜ Py_FalseпјҢеҲҷиҝ”еӣһ 0
        return 0;
    if (v == Py_None)           // еҰӮжһң v жҳҜ Py_NoneпјҢеҲҷиҝ”еӣһ 0
        return 0;
    else if (Py_TYPE(v)->tp_as_number != NULL &&
             Py_TYPE(v)->tp_as_number->nb_bool != NULL)     // еҰӮжһң v жҳҜдёҖдёӘж•°еӯ—зұ»еһӢпјҢеҲҷи°ғз”Ё nb_bool еҮҪж•°
        res = (*Py_TYPE(v)->tp_as_number->nb_bool)(v);
    else if (Py_TYPE(v)->tp_as_mapping != NULL &&
             Py_TYPE(v)->tp_as_mapping->mp_length != NULL)  // еҰӮжһң v жҳҜдёҖдёӘжҳ е°„зұ»еһӢпјҢеҲҷи°ғз”Ё mp_length еҮҪж•°
        res = (*Py_TYPE(v)->tp_as_mapping->mp_length)(v);
    else if (Py_TYPE(v)->tp_as_sequence != NULL &&
             Py_TYPE(v)->tp_as_sequence->sq_length != NULL) // еҰӮжһң v жҳҜдёҖдёӘеәҸеҲ—зұ»еһӢпјҢеҲҷи°ғз”Ё sq_length еҮҪж•°
        res = (*Py_TYPE(v)->tp_as_sequence->sq_length)(v);
    else
        return 1;
    /* if it is negative, it should be either -1 or -2 */
    return (res > 0) ? 1 : Py_SAFE_DOWNCAST(res, Py_ssize_t, int);
}

/* Perform a rich comparison, raising TypeError when the requested comparison
   operator is not supported. */
static PyObject *
do_richcompare(PyThreadState *tstate, PyObject *v, PyObject *w, int op)
{
    richcmpfunc f;
    PyObject *res;
    int checked_reverse_op = 0;

    if (!Py_IS_TYPE(v, Py_TYPE(w)) &&
        PyType_IsSubtype(Py_TYPE(w), Py_TYPE(v)) &&
        (f = Py_TYPE(w)->tp_richcompare) != NULL) {     // еҰӮжһң w жҳҜ v зҡ„еӯҗзұ»пјҢеҲҷи°ғз”Ё w зҡ„ tp_richcompare еҮҪж•°
        checked_reverse_op = 1;
        res = (*f)(w, v, _Py_SwappedOp[op]);
        if (res != Py_NotImplemented)
            return res;
        Py_DECREF(res);
    }
    if ((f = Py_TYPE(v)->tp_richcompare) != NULL) {   // и°ғз”Ё v зҡ„ tp_richcompare еҮҪж•°
        res = (*f)(v, w, op);
        if (res != Py_NotImplemented)
            return res;
        Py_DECREF(res);
    }
    if (!checked_reverse_op && (f = Py_TYPE(w)->tp_richcompare) != NULL) {   // и°ғз”Ё w зҡ„ tp_richcompare еҮҪж•°
        res = (*f)(w, v, _Py_SwappedOp[op]);
        if (res != Py_NotImplemented)
            return res;
        Py_DECREF(res);
    }
    /* If neither object implements it, provide a sensible default
       for == and !=, but raise an exception for ordering. */
    switch (op) {
    case Py_EQ:
        res = (v == w) ? Py_True : Py_False;
        break;
    case Py_NE:
        res = (v != w) ? Py_True : Py_False;
        break;
    default:
        _PyErr_Format(tstate, PyExc_TypeError,
                      "'%s' not supported between instances of '%.100s' and '%.100s'",
                      opstrings[op],
                      Py_TYPE(v)->tp_name,
                      Py_TYPE(w)->tp_name);
        return NULL;
    }
    Py_INCREF(res);
    return res;
}
